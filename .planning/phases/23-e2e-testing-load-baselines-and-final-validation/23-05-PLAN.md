<!-- CLI commands for data loading (from src/civpulse_geo/cli/__init__.py) -->

load-oa: geo-import load-oa <file.geojson.gz> --database-url <psycopg2_url>
  - Accepts .geojson.gz file
  - Upserts into openaddresses_points table
  - Infers state from filename pattern US_{state}_{county}_Addresses_...

load-nad: geo-import load-nad <file.zip> --state GA --database-url <psycopg2_url>
  - Accepts NAD_r21_TXT.zip file
  - Requires --state flag (GA for Bibb County)
  - COPY to temp table then INSERT...ON CONFLICT upsert into nad_points

load-macon-bibb: geo-import load-macon-bibb <file.geojson> --database-url <psycopg2_url>
  - Accepts uncompressed .geojson file (NOT .geojson.gz)
  - Upserts into macon_bibb_points table
  - State hardcoded to GA

<!-- Spell dictionary auto-rebuild (from src/civpulse_geo/main.py lifespan) -->
At startup, if spell_dictionary is empty AND staging tables have data,
rebuild_dictionary(conn) runs automatically. No manual step needed after data load.

<!-- Provider registration guards (from src/civpulse_geo/main.py) -->
Each provider checks its staging table count at startup:
  - openaddresses_points > 0 -> registers OA provider
  - nad_points > 0 -> registers NAD provider
  - macon_bibb_points > 0 -> registers Macon-Bibb provider
  - postgis_tiger_geocoder extension available -> registers Tiger provider
  - Census always registers (external API)
