    4. **Create `src/civpulse_geo/cli/parsers.py`** with three parser functions. Each returns `list[dict]` with uniform schema `{"properties": {...}, "geometry": {"coordinates": [lng, lat]}}`:

       **load_geojson(path: Path) -> list[dict]:**
       - Use `json.load()` (stdlib, same as seed.py)
       - Return `data["features"]` directly (GeoJSON features already have the correct schema)
       - Skip features missing geometry

       **load_kml(path: Path) -> list[dict]:**
       - Use `xml.etree.ElementTree` (stdlib, verified working in research)
       - KML namespace: `{"kml": "http://www.opengis.net/kml/2.2"}`
       - Find all `kml:Placemark` elements
       - Extract coordinates from `kml:coordinates` element: `lng,lat` (comma-separated, take first two)
       - Extract properties from `kml:SimpleData` elements: `name` attribute is key, text content is value
       - Return list of dicts matching GeoJSON schema

       **load_shp(path: Path) -> list[dict]:**
       - Use `fiona.open(path)` to read with CRS metadata
       - Check `src.crs` -- if not EPSG:4326, use `fiona.transform.transform_geom(src.crs, "EPSG:4326", feat["geometry"])` to reproject
       - Extract `lng, lat = geom_wgs84["coordinates"]`
       - Build properties dict from `feat["properties"]`
       - Return list of dicts matching GeoJSON schema

    2. **Implement the full `import_gis` command in `src/civpulse_geo/cli/__init__.py`:**
       ```python
       import typer
       from pathlib import Path
       from loguru import logger
       from sqlalchemy import create_engine, text

       from civpulse_geo.cli.parsers import load_geojson, load_kml, load_shp
       from civpulse_geo.normalization import canonical_key, parse_address_components
       from civpulse_geo.config import settings

       app = typer.Typer(help="CivPulse Geo CLI -- GIS data import tools.")

       # Bibb County GeoJSON/KML field mapping
       BIBB_FIELD_MAP = {
           "full_address": "FULLADDR",
           "city": "City_1",
           "zip_code": "ZIP_1",
           "state": "GA",  # Hardcoded for Bibb County, Georgia
       }


       @app.command("import")
       def import_gis(
           file: Path = typer.Argument(..., help="GeoJSON, KML, or SHP file to import"),
           database_url: str | None = typer.Option(None, "--database-url", envvar="DATABASE_URL_SYNC"),
           provider: str = typer.Option("bibb_county_gis", "--provider", help="Provider name for imported records"),
       ) -> None:
           """Import GIS address data as geocoding results for a provider."""
           db_url = database_url or settings.database_url_sync

           # Auto-detect format
           suffix = file.suffix.lower()
           if suffix == ".geojson":
               features = load_geojson(file)
           elif suffix == ".kml":
               features = load_kml(file)
           elif suffix == ".shp":
               features = load_shp(file)
           else:
               typer.echo(f"Unsupported file format: {suffix}. Use .geojson, .kml, or .shp", err=True)
               raise typer.Exit(1)

           typer.echo(f"Loaded {len(features)} features from {file.name}")

           engine = create_engine(db_url)
           stats = {"total": 0, "inserted": 0, "updated": 0, "skipped": 0, "errors": 0}

           with engine.connect() as conn:
               for i, feature in enumerate(features, 1):
                   stats["total"] += 1
                   try:
                       _import_feature(conn, feature, provider, stats)
                   except Exception as e:
                       stats["errors"] += 1
                       logger.warning(f"Feature {i}: {e}")

                   if i % 100 == 0:
                       conn.commit()
                       typer.echo(f"  Processed {i}/{len(features)}...")

               conn.commit()

           typer.echo(f"\nImport complete:")
           typer.echo(f"  Total:    {stats['total']}")
           typer.echo(f"  Inserted: {stats['inserted']}")
           typer.echo(f"  Updated:  {stats['updated']}")
           typer.echo(f"  Skipped:  {stats['skipped']}")
           typer.echo(f"  Errors:   {stats['errors']}")
       ```
