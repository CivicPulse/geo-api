    Add the following imports at the top of main.py:

    Add the handler:
    ```python
    @app.exception_handler(Exception)
    async def generic_exception_handler(request: Request, exc: Exception) -> JSONResponse:
        """Catch-all handler: converts unhandled exceptions to structured 500 JSON.

        Prevents raw tracebacks from reaching the client (STAB-01, STAB-02).
        Specific exception types (ValueError -> 404, ProviderError -> 422)
        are still caught by individual endpoints before this handler fires.
        """
        logger.error("Unhandled exception on {}: {}", request.url.path, exc)
        return JSONResponse(
            status_code=500,
            content={"detail": "Internal server error"},
        )
    ```

    Place this block AFTER `app = FastAPI(...)` and BEFORE the first `app.include_router(...)` call. The `logger` import from loguru is already present in main.py.

    **STAB-04: Per-provider try/except in legacy provider loop**

    In `src/civpulse_geo/services/geocoding.py`, find the `_legacy_geocode` method (or the method that iterates over providers in a `for` loop without try/except). There are TWO unguarded loops — the local provider loop (around lines 213-230) and the remote provider loop (around lines 249-310).

    Wrap each provider call inside both loops with a per-provider try/except:

    For the **local provider loop** (around lines 213-230):
    ```python
    for provider_name, provider in local_providers.items():
        try:
            provider_result = await provider.geocode(normalized, http_client=http_client)
            # ... existing result handling code
        except Exception as e:
            logger.warning("Legacy path: provider {} failed: {}", provider_name, e)
            continue
    ```

    For the **remote provider loop** (around lines 249-310):
    ```python
    for provider_name, provider in remote_providers.items():
        try:
            provider_result = await provider.geocode(normalized, http_client=http_client)
            # ... existing upsert and result handling code for this iteration
        except Exception as e:
            logger.warning("Legacy path: provider {} failed: {}", provider_name, e)
            continue
    ```

    Important: The `try/except` must wrap the ENTIRE body of each loop iteration (including the DB upsert for remote providers), not just the `provider.geocode()` call. This ensures any exception during result handling also degrades gracefully.

    ```python
    Verifies that unhandled exceptions from service layer are caught by the
    global exception handler in main.py and returned as structured 500 JSON.

    import pytest
    from unittest.mock import AsyncMock, patch

    from httpx import AsyncClient, ASGITransport

    from civpulse_geo.main import app
    from civpulse_geo.database import get_db


    @pytest.fixture
    def override_db():
        mock_session = AsyncMock()
        async def _override():
            yield mock_session
        app.dependency_overrides[get_db] = _override
        yield mock_session
        app.dependency_overrides.clear()


    @pytest.fixture
    def patched_app_state():
        app.state.http_client = AsyncMock()
        app.state.providers = {}
        app.state.validation_providers = {}
        app.state.spell_corrector = None
        app.state.fuzzy_matcher = None
        app.state.llm_corrector = None
        yield
        for attr in ["http_client", "providers", "validation_providers",
                      "spell_corrector", "fuzzy_matcher", "llm_corrector"]:
            try:
                delattr(app.state, attr)
            except AttributeError:
                pass


    @pytest.mark.asyncio
    async def test_geocode_unhandled_exception_returns_500_json(override_db, patched_app_state):
        """STAB-01: RuntimeError in geocode service returns structured 500, not raw traceback."""
        with patch("civpulse_geo.api.geocoding.GeocodingService") as MockService:
            MockService.return_value.geocode = AsyncMock(side_effect=RuntimeError("DB exploded"))
            transport = ASGITransport(app=app)
            async with AsyncClient(transport=transport, base_url="http://test") as client:
                resp = await client.post("/geocode", json={"address": "123 Main St"})
            assert resp.status_code == 500
            assert resp.json()["detail"] == "Internal server error"
            assert "application/json" in resp.headers.get("content-type", "")


    @pytest.mark.asyncio
    async def test_validate_unhandled_exception_returns_500_json(override_db, patched_app_state):

            MockService.return_value.validate = AsyncMock(side_effect=RuntimeError("DB exploded"))

            assert resp.json()["detail"] == "Internal server error"

    @pytest.mark.asyncio
    ```
