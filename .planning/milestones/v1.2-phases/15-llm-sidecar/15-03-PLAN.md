    **Step 1: Create scripts/ollama-entrypoint.sh** (per D-10, Research Pattern 5):
    ```bash
    #!/bin/bash
    set -euo pipefail

    # Start Ollama server in background
    /bin/ollama serve &
    OLLAMA_PID=$!

    # Wait for server to be ready
    echo "Waiting for Ollama server..."
    until curl -sf http://localhost:11434/api/tags > /dev/null 2>&1; do
      sleep 2
    done
    echo "Ollama server ready."

    # Pull model(s) — idempotent, skips if already cached in volume
    for MODEL in ${OLLAMA_MODELS//,/ }; do
      echo "Pulling model: $MODEL"
      ollama pull "$MODEL"
    done

    echo "Model(s) ready. Ollama serving."
    wait $OLLAMA_PID
    ```

    Make executable: `chmod +x scripts/ollama-entrypoint.sh`

    **Step 2: Add ollama service to docker-compose.yml** (per D-10, D-11, D-13):

    Add the ollama service between the db and api services. Add `ollama_data` to the volumes section. Update the api service to optionally depend on ollama (with condition: service_healthy) and add OLLAMA_URL and CASCADE_LLM_ENABLED env vars.

    The ollama service block:
    ```yaml
      ollama:
        image: ollama/ollama:latest
        entrypoint: ["/bin/bash", "/entrypoint.sh"]
        environment:
          - OLLAMA_MODELS=qwen2.5:3b
        volumes:
          - ollama_data:/root/.ollama
          - ./scripts/ollama-entrypoint.sh:/entrypoint.sh:ro
        mem_limit: 4g
        ports:
          - "11434:11434"
        healthcheck:
          test: ["CMD-SHELL", "curl -sf http://localhost:11434/api/tags || exit 1"]
          interval: 30s
          timeout: 10s
          retries: 10
          start_period: 120s
        profiles:
          - llm
    ```

    Note: Using Docker Compose profiles (`profiles: [llm]`) so ollama only starts when explicitly requested: `docker compose --profile llm up`. This aligns with CASCADE_LLM_ENABLED=false default — devs who don't need LLM don't download the 2GB model.

    Add to api service environment:
    ```yaml
      CASCADE_LLM_ENABLED: "${CASCADE_LLM_ENABLED:-false}"
      OLLAMA_URL: "http://ollama:11434"
    ```

    Add to volumes section:
    ```yaml
      ollama_data:
    ```

    ```yaml
    apiVersion: apps/v1
    kind: Deployment
    metadata:
      name: ollama
      labels:
        app: ollama
        app.kubernetes.io/part-of: civpulse-geo
    spec:
      replicas: 1
      selector:
        matchLabels:
          app: ollama
      template:
        metadata:
          labels:
            app: ollama
        spec:
          initContainers:
            - name: model-pull
              image: ollama/ollama:latest
              command: ["sh", "-c"]
              args:
                - |
                  /bin/ollama serve &
                  until curl -sf http://localhost:11434/api/tags > /dev/null 2>&1; do sleep 2; done
                  ollama pull qwen2.5:3b
                  kill %1
              volumeMounts:
                - name: ollama-data
                  mountPath: /root/.ollama
          containers:
            - name: ollama
              image: ollama/ollama:latest
              ports:
                - containerPort: 11434
              resources:
                requests:
                  memory: "2Gi"
                  cpu: "1"
                limits:
                  memory: "4Gi"
              volumeMounts:
                - name: ollama-data
                  mountPath: /root/.ollama
              readinessProbe:
                httpGet:
                  path: /api/tags
                  port: 11434
                initialDelaySeconds: 10
                periodSeconds: 10
                failureThreshold: 6
              livenessProbe:
                httpGet:
                  path: /api/tags
                  port: 11434
                initialDelaySeconds: 30
                periodSeconds: 30
                failureThreshold: 3
          volumes:
            - name: ollama-data
              persistentVolumeClaim:
    ```
